\begin{definition}
	Sei $\mu \in \R, \sigma^2 > 0$. Das Wahrscheinlichkeitsmaß $N(\mu, \sigma^2)$ auf $(\R, \borel(\R))$ mit Dichtefunktion
	\begin{align*}
		g_{\mu,\sigma^2}(x) := \frac{1}{\sqrt{2\pi\sigma^2}}e^{-\frac{(x-\mu)^2}{2\sigma^2}} \quad x \in \R
	\end{align*}
	heißt \begriff{Normalverteilung mit Parametern $\mu \und \sigma^2$}.
	Im Fall $\mu = 0,\sigma^2 = 1$, so heißt $N(0,1)$ \begriff{Standardnormalverteilung}.

\end{definition}
